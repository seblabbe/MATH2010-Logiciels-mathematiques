\frametitle{Boulier}
\begin{center}
\includegraphics[width=.3\linewidth]{images/220px-Abacus_6.png}
%Illustration de l'Encyclopedia Britannica (1875).
\includegraphics[width=.3\linewidth]{images/boulier-chinois.jpg}
\end{center}
\begin{itemize}
\item un des \alert{plus anciens instruments} d'aide au calcul de l'histoire de
l'humanité.
\item calcul des opérations élémentaires: \alert{additions},
\alert{soustractions}, \alert{multiplications} et \alert{divisions} et même la
\alert{racine} énième.
\item Utilisé par les Étrusques, les Grecs, les Égyptiens, les Indiens, les
Chinois et les Mexicains.
\end{itemize}
\url{http://fr.wikipedia.org/wiki/Boulier}
%\url{http://fr.wikipedia.org/wiki/Soroban}
\end{frame}

\begin{frame}
\frametitle{Chiffres arabes}
\begin{center}
\includegraphics[width=.3\linewidth]{images/math-crayon-papier.jpg}
\end{center}
\begin{itemize}
\item IIIe siècle av. J.-C.: numération décimale indienne
\item VIIe siècle: existence est mentionnée en Syrie
\item IXe siècle: empruntée par la civilisation arabe
\item 1202: \emph{Liber Abaci} (Le livre du calcul), Leonardo Fibonacci, Pise:
un traité sur les calculs et la comptabilité fondée sur le calcul décimal.
\item XVe siècle: les chiffres indo-arabes sont préférés aux chiffres romains en
Europe.
\end{itemize}
\url{http://fr.wikipedia.org/wiki/Chiffres_arabes}
\end{frame}
