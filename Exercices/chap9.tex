    \[
	\int \frac{\sin(x)\tan(x)+\cos(x)}{\sin(x)-\cos(x)^2}\,dx.
    \]
    %>>> integrate((sin(x)*tan(x)+cos(x))/(sin(x)-cos(x)**2), x)
